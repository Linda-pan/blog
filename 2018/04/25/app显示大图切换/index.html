app显示大图切换

在html后面增加js

<script>
if(window.capvisionJSBridge){
    function showImage(index,urls){
        window.capvisionJSBridge.trigger(
            JSON.stringify({
                action:'showImage',
                data:{
                    index:index,
                    urls:urls
                }
            })
        );
    }
var $=document.querySelectorAll.bind(document);
var $images=$('img');
var sources=[];
for(i=0;i<$images.length;i++){
    var $item=$images[i];
    $item.onclick=function(event){
        var src=event.target.getAttribute('src');
        showImage(sources.indexOf(src),sources)};
        sources.push($item.getAttribute('src'));
    }
}
</script>
